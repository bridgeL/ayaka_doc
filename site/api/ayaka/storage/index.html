    def plugin_path¶

    获取路径 /../*names

    无参数

        def plugin_path(self, *names):
        '''获取路径 <create_app_file>/../*names'''
        names = [str(name) for name in names]
        path = Path(self.app.path.parent, *names)
        return AyakaPath(path)

    def group_path¶

    获取路径 data/groups////*names

    无参数

        def group_path(self, *names):
        '''获取路径 data/groups/<bod_id>/<group_id>/<app.name>/*names'''
        names = [
            "data", "groups",
            self.app.bot_id,
            self.app.group_id,
            self.app.name,
            *names
        ]
        names = [str(name) for name in names]
        path = Path(*names)
        return AyakaPath(path)

    def file¶

    参数表：

    • name
    • default
        def file(self, name, default=None):
        path = self.path / str(name)
        file = AyakaFile(path)
        if not path.exists() and default is not None:
            file.save(default)
        return file

    def json¶

    参数表：

    • name
    • default
        def json(self, name, default={}):
        path = self.path / str(name)
        path = path.with_suffix(".json")
        file = AyakaJsonFile(path)
        if not path.exists():
            file.save(default)
        return file
